147

153 
154  if (retval < 0) {
155  fprintf(stderr, "Error selecting on stdin\n");
156  return retval;
157  } else if (retval == 0) {
158  ; /* Timeout. Ignore. */
159  } else if (FD_ISSET(fileno(stdin), &fds)) {
160  rl_callback_read_char();
161  }
162  if (ready_to_submit) {

164  if (cur_input != NULL) {
165  scheme_load_string(&sc, cur_input);
166  printf("\n");
167  free(cur_input);

169  }
